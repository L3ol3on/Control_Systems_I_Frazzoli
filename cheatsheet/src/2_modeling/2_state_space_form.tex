        \begin{align*}
            \begin{cases}
                \dot{x}(t) = f(x(t), u(t), d(t))\\
                y(t) = f(x(t), u(t), d(t))
            \end{cases}\\
        \end{align*}
    \end{minipage}
    \begin{minipage}{0.49 \linewidth}

        \begin{align*}

                \dot{x}(t) = Ax(t) + Bu(t)\\
                y(t) = Cx(t) + Du(t)
            \end{cases}\\
        \end{align*}
    \end{minipage}
